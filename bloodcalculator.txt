def interface():
    print("blood calculator")
    while True:
        print("Options:")
        print("1 = HDL")
        print("2 = LDL")
        print("3 = Quit")
        print("4 = Cholesterol")
        choice = input("Select an option: ")
        if choice == "3":
            return
        elif choice == "1":
            HDL_driver()
        elif choice == "2":
            LDL_driver()
        elif choice == "4":
            Chol_driver()
    print("Program ending")
